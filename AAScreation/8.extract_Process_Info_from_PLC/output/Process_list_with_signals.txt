  -SMC Process Crane_MoveBase
    -Ref Extends: ControlPrimitiveBase
    -SMC VAR:
      -Prop m_ptrCrane: POINTER TO Crane;
    -SMC METHODS:
      -SMC Abort:
          -Prop m_ptrCrane^.METH_CarrierStop();
          -Prop Abort := TRUE;
      -SMC ExecuteInternal:
          -Prop angle := m_st^.GetNumeric(EProcessVariables.Crane_Angle);
          -Prop diff := TargetAngle - angle;
          -Prop logic:
              IF ABS(diff) < tolerance THEN
              m_ptrCrane^.METH_CarrierStop();
              ExecuteInternal := TRUE;
              ELSIF diff < 0 THEN
              m_ptrCrane^.METH_CarrierTurnClockwise();
              ELSE
              m_ptrCrane^.METH_CarrierTurnCounterClockwise();
              END_IF

  -SMC Process LSC_SearchWorkpiece
    -Ref Extends: ControlPrimitiveBase
    -SMC VAR:
      -Prop m_passby: BOOL;
      -Prop m_postcondition: LSC_ProductType >= 0;
      -Prop m_precondition: TRUE;
      -Prop m_ptrLSC: POINTER TO LargeSortingConveyor;
      -Prop m_step: INT := 0;
      -Prop m_timer: TON;
    -SMC METHODS:
      -SMC Abort:
          -Prop m_ptrLSC^.METH_ConveyorStop();
          -Prop m_ptrLSC^.METH_Ramp1Pusher_Retract();
          -Prop m_ptrLSC^.METH_Ramp2Pusher_Retract();
          -Prop m_ptrLSC^.METH_Switch_Retract();
          -Prop Abort := TRUE;
      -SMC ExecuteInternal:
        -SMC step 0 search in front of presence sensor:
          -Prop m_ptrLSC^.METH_ConveyorForward();
          -Prop m_timer(IN := TRUE, PT := T#600MS);
          -Prop logic 1:
              IF m_timer.Q THEN	//empty in front of presence sensor;
              m_timer(IN := FALSE);
              m_ptrLSC^.METH_ConveyorStop();
              m_step := 1;
              END_IF
          -Prop logic 2:
              IF m_ptrLSC^.WPAtStart THEN //Workpiece detected;
              m_timer(IN := FALSE);
              m_ptrLSC^.METH_ConveyorStop();
              m_step := 2;
              END_IF
        -SMC step 1 search behind presence sensor:
          -Prop m_ptrLSC^.METH_ConveyorBackward();
          -Prop m_timer(IN := TRUE, PT := T#3000MS);
          -Prop logic 1:
              IF m_timer.Q THEN	//empty behind presence sensor;
              m_timer(IN := FALSE);
              m_ptrLSC^.METH_ConveyorStop();
              m_step := 4;
              END_IF
          -Prop logic 2:
              IF m_ptrLSC^.WPAtStart THEN //Workpiece detected;
              m_timer(IN := FALSE);
              m_ptrLSC^.METH_ConveyorStop();
              m_step := 2;
              END_IF
        -SMC step 2 workpiece found, identify workpiece  (300ms):
          -Prop m_ptrLSC^.METH_ConveyorForward();
          -Prop m_timer(IN := TRUE, PT := T#300MS);
          -Prop logic 1:
              IF m_timer.Q THEN	//workpiece at inductive and optical sensor;
              m_timer(IN := FALSE);
              m_ptrLSC^.METH_ConveyorStop();
              m_step := 3;
              END_IF
        -SMC step 3 move workpiece to presence sensor:
          -Prop m_ptrLSC^.METH_ConveyorBackward();
          -Prop logic 1:
              IF m_ptrLSC^.WPAtStart THEN
              m_ptrLSC^.METH_ConveyorStop();
              m_step := 0;
              ExecuteInternal := TRUE;
              END_IF
        -SMC step 4 no workpiece found; empty conveyor:
          -Prop m_ptrLSC^.METH_ConveyorForward();
          -Prop m_timer(IN := TRUE, PT := T#4000MS);
          -Prop logic 1:
              IF m_timer.Q THEN	//conveyor is empty;
              m_timer(IN := FALSE);
              m_ptrLSC^.METH_ConveyorStop();
              m_step := 0;
              ExecuteInternal := TRUE;
              END_IF
      -SMC FB_init:
          -Prop m_ptrLSC := i_ptrLSC;
          -Prop m_id := EPrimitives.LSC_SearchWorkpiece;
          -Prop m_timeout := T#200S;
          -Prop Precondition := m_precondition;
          -Prop Postcondition := m_postcondition;
    -SMC Call Skill:
      -Ref Skill1: LargeSortingConveyor_Skill_ConveyorBackward
      -Ref Skill2: LargeSortingConveyor_Skill_ConveyorForward
      -Ref Skill3: LargeSortingConveyor_Skill_ConveyorStop
      -Ref Skill4: LargeSortingConveyor_Skill_Ramp1Pusher_Retract
      -Ref Skill5: LargeSortingConveyor_Skill_Ramp2Pusher_Retract
      -Ref Skill6: LargeSortingConveyor_Skill_Switch_Retract
    -SMC Control Variable:

  -SMC Process Stamp_StampWorkpiece
    -Ref Extends: ControlPrimitiveBase
    -SMC VAR:
      -Prop m_postcondition: Stamp_WorkpieceState = 2;
      -Prop m_precondition: Stamp_WPAtStamp = TRUE;
      -Prop m_ptrStamp: POINTER TO Stamp;
      -Prop m_step: INT := 0;
      -Prop t_Stamp : TON;
    -SMC METHODS:
      -SMC Abort:
          -Prop m_ptrStamp^.m_stampingCylinder.METH_Retract();
          -Prop m_ptrStamp^.m_slidingCylinder.METH_Extend();
          -Prop Abort := TRUE;
      -SMC ExecuteInternal:
          -Prop ExecuteInternal := FALSE;
        -SMC step 0 take in wp:
          -Prop m_ptrStamp^.m_slidingCylinder.METH_Retract();
          -Prop logic 1:
              IF m_ptrStamp^.m_slidingCylinder.Retracted THEN
              m_step := 1;
              END_IF
        -SMC step 1 stamp wp:
          -Prop m_ptrStamp^.StampPressure := GetPressure(i_params);
          -Prop m_ptrStamp^.m_stampingCylinder.METH_Extend();
          -Prop t_Stamp(IN:=TRUE, PT:=T#1500MS);
          -Prop logic 1:
              IF t_Stamp.Q THEN
              IF m_ptrStamp^.StampPressure >= m_pressureVal.Value.Numeric THEN
              m_step := 2;
              t_Stamp(IN:=FALSE);
              END_IF
              END_IF
        -SMC step 2 retract stamp:
          -Prop m_ptrStamp^.m_stampingCylinder.METH_Retract();
          -Prop logic 1:
              IF m_ptrStamp^.m_stampingCylinder.Retracted THEN
              m_step := 3;
              END_IF
        -SMC step 3 provide wp:
          -Prop m_ptrStamp^.m_slidingCylinder.METH_Extend();
          -Prop logic 1:
              IF m_ptrStamp^.m_slidingCylinder.Extended THEN
              m_step := 0;
              ExecuteInternal := TRUE;
              END_IF
      -SMC FB_init:
          -Prop m_ptrStamp := i_ptrStamp;
          -Prop m_id := EPrimitives.Stamp_StampWorkpiece;
          -Prop m_timeout := T#200S;
          -Prop m_paramDefinition := 'pressure_white:INT;pressure_black:INT;pressure_metal:INT;';
          -Prop Precondition := m_precondition;
          -Prop Postcondition := m_postcondition;
      -SMC GetPressure:
          -Prop logic:
              CASE LREAL_TO_UINT(m_st^.GetNumeric(EProcessVariables.Stamp_ProductType)) OF
              EProductTypes.white:;
              m_pressureVal := i_params.Retrieve('pressure_white');
              GetPressure := LREAL_TO_INT(m_pressureVal.Value.Numeric);
              EProductTypes.black:;
              m_pressureVal := i_params.Retrieve('pressure_black');
              GetPressure := LREAL_TO_INT(m_pressureVal.Value.Numeric);
              EProductTypes.metal:;
              m_pressureVal := i_params.Retrieve('pressure_metal');
              GetPressure := LREAL_TO_INT(m_pressureVal.Value.Numeric);
              ELSE
              GetPressure := 0;
              END_CASE
    -SMC Call Skill:
      -Ref Skill1: slidingCylinder_Skill_Extend
      -Ref Skill2: slidingCylinder_Skill_Retract
      -Ref Skill3: stampingCylinder_Skill_Extend
      -Ref Skill4: stampingCylinder_Skill_Retract
